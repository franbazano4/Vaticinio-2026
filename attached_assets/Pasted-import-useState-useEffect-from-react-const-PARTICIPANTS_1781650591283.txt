  GROUP_MATCHES.filter(m=>m.group===g).forEach(m=>{
    const res=results[m.id]; if(!res) return;
    const [rh,ra]=res;
    st[m.home].pj++;st[m.away].pj++;
    st[m.home].gf+=rh;st[m.home].gc+=ra;
    st[m.away].gf+=ra;st[m.away].gc+=rh;
    if(rh>ra){st[m.home].g++;st[m.home].pts+=3;st[m.away].p++;}
    else if(ra>rh){st[m.away].g++;st[m.away].pts+=3;st[m.home].p++;}
    else{st[m.home].e++;st[m.home].pts++;st[m.away].e++;st[m.away].pts++;}
  });
  return teams.map(t=>({team:t,...st[t],dg:st[t].gf-st[t].gc})).sort((a,b)=>b.pts-a.pts||b.dg-a.dg||b.gf-a.gf);
}

function calcAllThirds(results){
  return Object.keys(GROUPS).map(g=>{
    const st=calcGroupStandings(g,results);
    return st[2]?{...st[2],group:g}:null;
  }).filter(Boolean).sort((a,b)=>b.pts-a.pts||b.dg-a.dg||b.gf-a.gf);
}

function calcPts(pred,real){
  if(!pred||!real) return null;
  const [ph,pa]=pred,[rh,ra]=real;
  const rs=rh>ra?1:rh<ra?-1:0,ps=ph>pa?1:ph<pa?-1:0;
  if(ph===rh&&pa===ra) return 4;
  if(ps===rs) return 2;
  return 0;
}

function PtsBadge({pred,real}){
  const pts=calcPts(pred,real);
  if(pts===null) return null;
  const col=pts===4?C.green:pts===2?C.yellow:C.red;
  return <span style={{fontSize:10,fontWeight:700,padding:"1px 5px",border:`1px solid ${col}`,color:col,borderRadius:2,marginLeft:4}}>{pts>0?`+${pts}`:0}</span>;
}
